== macOS

=== host

Start server::
[source, zsh]
----
pulseaudio --load=module-native-protocol-tcp --exit-idle-time=-1 --daemon
pulseaudio --check -v
pacmd list-sinks | grep -e 'name:' -e 'index:' -e 'card:'
----

Add environment variable `HOST_IFADDR_EN0` to `.zshrc` (adapt to shell used)::
[source, zsh]
----
HOST_IFADDR_EN0=$(ipconfig getifaddr en0)
----

"remoteEnv": {

"mounts": ["source=${localEnv:HOME}/.config/pulse,target=/home/vscode/.config/pulse,type=bind,consistency=cached"]
